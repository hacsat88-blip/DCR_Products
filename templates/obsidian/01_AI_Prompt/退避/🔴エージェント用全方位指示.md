# 1. Agent Tools
agent_update_plan(goal: str, phases: list)
agent_advance_phase(from_phase_id: int, to_phase_id: int)
agent_end_task()
agent_schedule_task(name: str, prompt: str, schedule_type: str, cron: str = None, interval_seconds: int = None)

# 2. Message Tools
# Send message or ask user for input.
message_notify_user(text: str, attachments: list = [])
message_ask_user(text: str, options: list = [], enforce_choice: bool = False) # Use enforce_choice=True for strict options.

# 3. Shell & Filesystem Tools
# Execute shell commands.
shell_exec(command: str, work_dir: str = ".", reason: str = None) # Use 'reason' to explain why.
shell_view(session_id: str)
shell_wait(session_id: str, timeout: int = None)
shell_input(session_id: str, data: str)

# 4. State Check Tools
# Check status of files, processes, or network without using shell_exec.
check_path(path: str, check_type: str) # types: 'exists', 'is_file', 'is_dir', 'is_empty'. Returns bool.
check_process(pid: int = None, name: str = None) # type: 'is_running'. Returns bool.
check_network(host: str, port: int) # type: 'is_port_open'. Returns bool.

# 5. Composite Tools
# High-level, multi-step operations.
git_setup(repo_url: str, target_dir: str, install_cmd: str = None, reason: str = None) # Clones repo, cds into it, and runs install command.
archive(action: str, source_paths: list, dest_path: str, reason: str = None) # actions: 'compress', 'decompress'. format: 'zip', 'tar.gz'.
